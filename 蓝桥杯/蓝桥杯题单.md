```java
                num=0;

                sum+=num;
                num=0;

                num=num*10+(ch-'0');
            }
        }

    }
}
```

```java
import java.util.*;

public class Main{

        static final int N=20007;
        static int[] primes = new int[N];
        static boolean[] st = new boolean[N]; //false 代表是素数

        public static void main(String [] args){
            Scanner sc=new Scanner(System.in);
            int n =sc.nextInt();

            int cnt=0;
            for(int i=2;i<=n;i++){
                if(st[i]) continue; //如果是true直接跳过
                //false 的情况
                primes[cnt++]=i;
                for(int j=i+i;j<=n;j+=i) st[j]=true;
            }
            //筛完之后 素数都存在primes这个数组中
             for (int i = 0; i < cnt; i++) {
            // 中层循环 j 从 i 到 cnt-1，这样能保证第一个素数不会变大，同时让第二个素数尽可能小
            for (int j = i; j < cnt; j++) {
                // 内层循环 k 从 j 到 cnt-1，这样能保证前两个素数不变时，第三个素数尽可能小
                for (int k = j; k < cnt; k++) {
                    int sum = primes[i] + primes[j] + primes[k];
                    if (sum == n) {
                        System.out.printf("%d %d %d", primes[i], primes[j], primes[k]);
                        return;
                    }
                  // 优化：如果和已经大于n，可以提前结束内层循环
                    if (sum > n) {
                        break;
                    }
                }
            }
        }
        }
    }
```

<img src="assets/image-20260408172615678.png" alt="image-20260408172615678" style="zoom:50%;" />

```java
import java.util.Scanner;

/**
 * 计算给定区间 [l, r] 内素数的个数。
 * 使用分段筛法 (Segmented Sieve of Eratosthenes)。
 */
public class Main {

    private static final int MAX_SIZE = 1000001;
    private static final int SQRT_MAX_R = 50000; // 根号R的最大值，用于预处理小素数

    // 预处理数组，存储小于SQRT_MAX_R的素数
    private static int[] prime = new int[MAX_SIZE];
    private static boolean[] vis = new boolean[MAX_SIZE]; // 用于标记是否为合数
    private static int cnt = 0; // 记录预处理素数的个数

    /**
     * 线性筛法预处理出 sqrt(SQRT_MAX_R) 范围内的所有素数
     * 注意：原C++代码中的Gprime函数实际上是标准的埃拉托斯特尼筛法，
     * 而不是线性筛。这里保持原意，实现标准筛法。
     */
    private static void generatePrimes() {
        // 初始化 vis 数组，false 表示可能是素数
        for (int i = 2; i <= SQRT_MAX_R; i++) {
            if (!vis[i]) {
                prime[++cnt] = i;
            }
            // 标记 i 的所有倍数为合数
            for (int j = 1; j <= cnt && i * prime[j] <= SQRT_MAX_R; j++) {
                vis[i * prime[j]] = true;
                // 标准筛法，不break
            }
        }
    }


   public static int countPrimesInRange(long L, long r) {
    if (r < 2) {
        return 0; // 如果r小于2，则区间内不可能有素数
    }
    
    // 特判L=1的情况，因为1不是素数，将其视为2
    if (L == 1) {
        L = 2; // 1不是素数，所以从2开始统计
    }

    // 如果是首次调用，先生成预处理的素数表
    if (cnt == 0) {
        generatePrimes(); // 只需执行一次预处理
    }

    // 创建一个布尔数组来标记区间 [l, r] 内的数是否为合数
    // 数组大小为 r - l + 1，索引 0 对应 l，索引 r-l 对应 r
    boolean[] segVis = new boolean[(int)(r - L + 1) + 1]; // 创建一个大小为区间长度的辅助数组

    // 遍历预处理得到的所有素数，用它们去筛掉区间 [l, r] 内的合数
    for (int i = 1; i <= cnt; i++) { // 遍历所有预处理好的小素数
        long p = prime[i]; // 当前用来筛的素数 p
        // 计算从哪个数开始标记，即大于等于l的最小的p的倍数
        // (l + p - 1) / p 得到大于等于 l/p 的最小整数，再乘以p
        long start = ((L + p - 1) / p) * p;
        
        // 确保起始点不小于 p*p (因为小于p*p的p的倍数已被更小的素数筛掉)
        // 这个判断是为了避免重复标记
        if (start < p * p) {
            start = p * p;
        }

        // 从 start 开始，以 p 为步长，标记区间 [l, r] 内的合数
        for (long j = start; j <= r; j += p) { // 从 start 开始，每次加 p
             // j >= l 确保索引不越界，虽然理论上循环条件已保证
             if (j >= L) { // 这个判断是冗余的，因为循环条件保证了 j <= r 且 start >= l 或 p*p
                 segVis[(int)(j - L)] = true; // 标记 j 是合数。 j-l 将 j 映射到 segVis 数组的索引上
             }
        }
    }

    // 统计区间 [l, r] 内未被标记的数（即素数）的个数
    int ans = 0;
    for (int i = 0; i <= r - L; i++) { // 遍历辅助数组 segVis
        if (!segVis[i]) { // 如果 segVis[i] 是 false，说明原始数字 l+i 没有被标记为合数，即它是素数
            ans++; // 计数器加一
        }
    }

    return ans; // 返回素数个数
}

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long L = scanner.nextLong();
        long r = scanner.nextLong();

        int result = countPrimesInRange(L, r);
        System.out.println(result);

        scanner.close();
    }
}
```

```java
import java.util.*;
//这个算法比暴力优化了,每个数的最小质因数,不需要再从2开始遍历,直接从最小质因数开始遍历
//比如49=7*7; 这个算法直接从7开始算,而我原来的从2依次遍历,会导致超时
public class Main{
    
    static int [] prime= new int [10000010];
    static int [] cnt= new int [10000010];
    //prime存最小质因数，cnt存质因数个数
    public static void aishishai(int m) {
    	prime[1] = 1;
        cnt[1] = 0;
    	for (int i = 2; i <= m; i++) {
    		if (prime[i] == 0) { //i为质数
    			prime[i] = i; //标记
    			for (int j = i * 2; j <= m; j += i) { //标记i的倍数
    				if (prime[j] == 0) { //
    					prime[j] = i; //标记
    				}
    			}
    		}
    		int x = i, cnts = 0;
    		while (x != 1) {
    			int p = prime[x]; //x的最小质因数
    			//除尽所有p
    			while (x % p == 0) {
    				cnts++;
                    x /= p; //计算结果并缩小范围
    			}
    		}
    		cnt[i] = cnts; //结果存入cnt数组
    	}
    }
    public static void main(String [] args) {
        	Scanner sc =new Scanner(System.in);
            int n=sc.nextInt();
            int m=sc.nextInt();
        	
        	aishishai(m); //预处理
        	int ans = 0;
        	for (int i = n; i <= m; i++) {
        		ans = Math.max(ans, cnt[i]); //求出n~m最多质因数的个数
        	}
        	System.out.print(ans);
        	sc.close();
    }
}
```

```java
public class Main{
    static final int N=100005;
    static int [] a=new int [N]; 
    static int [] d=new int [N];

    static StreamTokenizer in = new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));

    public static int nextInt() throws IOException{
        in.nextToken();
        return (int)in.nval;
    }

    public static void main(String [] args) throws IOException{
        int n=nextInt();
        for(int i=1;i<=n;i++){
            a[i]=nextInt();
        }
        //构造差分数组:
        for(int i=1;i<=n;i++){
            d[i]=a[i]-a[i-1];
        }
        long ans=0;
       
        //我们要把原数组a所有数都变成1,那么对应的差分数组就要都变成0,除了
        //最后一个数为1
        //对差分数组每次操作时,对一个数-1,一定对应另一个数+1
        //除非是最后一个,只需要-1,所以每次操作就是一个数-1 另一个+1或者只-1

        //所以可以转换为只计算 差分数组的正数和,最后再-1即可
        for(int i=1;i<=n;i++){

    }
```

### [自建OJ：闪耀的灯光](http://47.121.118.174/p/617)



<img src="assets/image-20260528112758415.png" alt="image-20260528112758415" style="zoom:50%;" /><img src="assets/image-20260528112753646.png" alt="image-20260528112753646" style="zoom:50%;" />

```java
/*
注意这道题有坑,我最初看到这道题要进行区间操作,直接去用差分数组了
但是这道题t次操作,每次操作前都会恢复初始亮度,而用差分数组会改变原数组
而且这道题肯定要用到前缀和数组,原数组改变,前缀和数组也要改变,
我们改变数组之后,每次都要再求原数组,再求前缀和数组 O(tmn) 会超时

这道题可以只用前缀和数组,初始化前缀和数组之后,我们每次操作,区间内所有灯泡都会减少同样的亮度
因为每次操作前都会还原亮度,所以我们只需要用前缀和数组求出区间和后
我们再减去这区间内减少的亮度和即可(d*(x2-x1+1)*(y2-y1+1)) d=k%(c+1)
因为操作c次后,下一次操作就会还原亮度,所以周期是 c+1,所以最后就等同于
只操作了d次
*/

import java.util.*;
import java.io.*;

public class Main{

    static int [][] a=new int [305][305];
    static long [][] s=new long [305][305];


    static StreamTokenizer in =new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));

    public static int nextInt() throws IOException{
        in.nextToken();
        return (int) in.nval;
    }

    public static void main(String [] args)throws IOException{
        int n=nextInt();
        int m=nextInt();
        int c=nextInt();

        //初始化 每个矩阵的初始亮度
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                a[i][j]=nextInt();
                //初始化前缀和数组
                s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+a[i][j];
            }
        }

        int t=nextInt();
        //t次操作:
        StringBuilder sb =new StringBuilder();
        for(int i=1;i<=t;i++){
            int x1=nextInt();
            int y1=nextInt();
            int x2=nextInt();
            int y2=nextInt();
            int k=nextInt();
            int d=k%(c+1);
            int cnt=(x2-x1+1)*(y2-y1+1);
            //只需要用前缀和数组求出区间和,然后呢,再减去每个灯泡减掉的亮度
            long sum=s[x2][y2]-s[x2][y1-1]-s[x1-1][y2]+s[x1-1][y1-1];
            sum-=cnt*d;
            sb.append(sum+"\n");
        }
        System.out.print(sb);

    }
}
```
