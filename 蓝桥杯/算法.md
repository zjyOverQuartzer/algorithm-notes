```java
             for (int i = 0; i < cnt; i++) {

                    }

                    if (sum > n) {
```

```java
public class 从1到n的最小公倍数{

    static final long mod = 1000000007;
    static HashMap<Integer,Integer> mp=new HashMap<>();
    static long ans=1;

    public static void main(String [] args){
        Scanner sc=new Scanner(System.in);
        int n = sc.nextInt();

        for(int i=2;i<=n;i++) divide(i);

        //质因数分解之后 遍历map集合
        for(Map.Entry<Integer,Integer> e:mp.entrySet()){
            int x=e.getKey();
            int y=e.getValue();
            for(int i=1;i<=y;i++) ans=ans*x%mod;//这样比乘完之后再取余更快
        }

        System.out.println(ans);

    }

    public static void divide(int num){
        //先进行 质因数分解
        for(int i=2;i*i<=num;i++){
            int cnt=0;
            while(num%i==0){
                cnt++;
                num/=i;
            }
            //将质因数的指数最大放入map集合中
            if(cnt>0){
                mp.put(i,Math.max(mp.getOrDefault(i,0),cnt));
            }
        }
        if(num>1){
            //对剩下的数字进行处理
            mp.put(num,Math.max(mp.getOrDefault(num,0),1));
        }
    }
}
```

### [洛谷：最大正方形](https://www.luogu.com.cn/problem/P1387)

<img src="assets/image-20260325160302647.png" alt="image-20260325160302647" style="zoom:50%;" />

```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 读取矩阵的行数和列数
        int n = scanner.nextInt();
        int m = scanner.nextInt();

        // 创建矩阵
        int[][] matrix = new int[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                matrix[i][j] = scanner.nextInt();
            }
        }
        scanner.close(); // 关闭Scanner以避免警告

        // dp[i][j] 表示以 (i,j) 为右下角的最大正方形的边长
        int[][] dp = new int[n][m];
        int maxSide = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (matrix[i][j] == 1) {
                    if (i == 0 || j == 0) {
                        // 如果在第一行或第一列，则最大正方形边长为1
                        dp[i][j] = 1;
                    } else {
                        // 否则，状态转移方程
                        dp[i][j] = Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1;
                    }
                    // 更新全局最大边长
                    maxSide = Math.max(maxSide, dp[i][j]);
                } else {
                    // 当前位置为0，则不可能是任何全1正方形的右下角
                    dp[i][j] = 0;
                }
            }
        }

        System.out.println(maxSide);
    }
}
```

### [洛谷：三国游戏](https://www.luogu.com.cn/problem/P13887)

```java
import java.util.*;

public class Main {
    private static int[] a = new int[100010];
    private static int[] b = new int[100010];
    private static int[] c = new int[100010];
    private static int[] s = new int[100010]; // s为贡献值数组
    private static int n;

    
    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();

        for(int i=1;i<=n;i++) a[i]=sc.nextInt();
        for(int i=1;i<=n;i++) b[i]=sc.nextInt();
        for(int i=1;i<=n;i++) c[i]=sc.nextInt();

        int result = Math.max(Math.max(f(0),f(1)),f(2));
        System.out.println(result);
    }
     public static int f(int t) {
        for (int i = 1; i <= n; i++) {
            if (t == 0) { // 魏国
                s[i] = a[i] - b[i] - c[i];
            } else if (t == 1) { // 蜀国
                s[i] = b[i] - a[i] - c[i];
            } else if (t == 2) { // 吴国
                s[i] = c[i] - b[i] - a[i];
            }
        }
        // 排序。注意：Java数组是0索引的，所以我们排序从索引1开始，到n结束
        Arrays.sort(s, 1, n + 1);

        long sum = 0;
        int ans = 0;
        // 从最大值开始累加
        for (int i = n; i >= 1; i--) {
            sum += s[i]; // 累计和
            if (sum > 0) { // sum>0，代表事件有效
                ans++;
            } else { // 无效退出
                //说明即使加上了当前这个最大的贡献值，也无法让国家获胜。
                //后面的循环都是无效的了
                break;
            }
        }
        if (ans == 0) { // 不能分出胜负
            return -1;
        } else {
            return ans;
        }
    }
}
```

## 并查集:

并查集是解决不相交集合并问题的一种数据结构，简称为 `dsu`（disjoint set union）。

<img src="assets/image-20260508194323815.png" alt="image-20260508194323815" style="zoom:50%;" />

```java
import java.io.*;
import java.util.*;

public class Main{
    static final int N=100000+5;
    static int[] f=new int[N];

    public static int find(int x){
        if(x==f[x]) return x;
        return f[x]=find(f[x]);
    } 

    public static void merge(int x,int y){
        int fx=find(x);
        int fy=find(y);
        if(fx!=fy) f[fx]=f[fy];
    }
    public static boolean same(int x,int y){
        int fx=find(x);
        int fy=find(y);
        return fx==fy;
    }

    public static void main(String[] args)throws Exception{
        BufferedReader br=new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st=new StringTokenizer(br.readLine());
        int n=Integer.parseInt(st.nextToken());
        int m=Integer.parseInt(st.nextToken());

        StringBuilder sb=new StringBuilder();
        for(int i=1;i<=m;i++){
            st=new StringTokenizer(br.readLine());
            String op=st.nextToken();
            int x=Integer.parseInt(st.nextToken());
            int y=Integer.parseInt(st.nextToken());
            if(op.equals("M")) merge(x,y);
```
